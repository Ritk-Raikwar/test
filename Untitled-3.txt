class UserMainCode(object):
    @classmethod
    def userIdGeneration(cls, input1, input2, input3, input4):

        first_name = input1.strip()
        last_name = input2.strip()
        pin = str(input3)
        N = input4

        smaller_name , longer_name = None ,None
        #step 1
        if len(first_name) < len(last_name):
            smaller_name = first_name
            longer_name = last_name
        elif len(first_name) > len(last_name):
            smaller_name = last_name
            longer_name = first_name
        else:
            
            for i in range(len(first_name)):
                if first_name[i] < last_name[i]:
                    smaller_name = first_name
                    longer_name = last_name
                    break
                elif first_name[i] > last_name[i]:
                    smaller_name = last_name
                    longer_name = first_name
                    break
        

        #step 2
        user_id = longer_name[0] + smaller_name + pin[N-1] + pin[len(pin)-N]

        #step 3
        answer = ""
        for i in user_id:
            if i.isupper():
                answer +=  i.lower()
            elif i.islower():
                answer += i.upper()
            else:
                answer += i

        return answer
